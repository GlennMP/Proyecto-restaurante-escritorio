call lista_usuarios

-- crear procedimiento
CREATE PROCEDURE `lista_usuarios`()
BEGIN
select d.CODIGO_EMPLEADO,d.USUARIO,d.NOMBRE_EMPLEADO,d.CONTRASEÑA_EMPLEADO,d.TELEFONO_EMPLEADO,d.DIRECCION_EMPLEADO,v.NOMBRE_CARGO from EMPLEADO d 
join CARGO v on d.ID_CARGO=v.ID_CARGO where ESTADO_EMPLEADO="a";
END

-- filtrado por nombre
CREATE PROCEDURE `listapornombre` (NOMBRE VARCHAR(50))
BEGIN
select d.CODIGO_EMPLEADO,d.USUARIO,d.NOMBRE_EMPLEADO,d.CONTRASEÑA_EMPLEADO,d.TELEFONO_EMPLEADO,d.DIRECCION_EMPLEADO,v.NOMBRE_CARGO from EMPLEADO d 
join CARGO v on d.ID_CARGO=v.ID_CARGO where NOMBRE_EMPLEADO = NOMBRE;
END
